class BaseBashCommand:
    filter_text = "something"
    def query(self):
        return "cat file.txt"

    def filter(self, text):
        return f"| grep \"{text}\""

    def sort(self):
        return "| sort"

    def make(self):
        query = (
            self.query() +
            self.filter(self.filter_text) +
            self.sort())
        return exec(query)

class MyCommand(BaseBashCommand):
    filter_text = "something"

class NewAccountsViewSet(ListModelMixin, GenericViewSet, DateRangeMixin):
    queryset = (
        User.objects.filter(
            permission_level=User.ADVERTISER,
            advertiser_profile__advertiser_id=AdvertiserProfile._meta.get_field(
                "advertiser_id"
            ).get_default(),
        )
        .order_by("-date_joined")
        .select_related("advertiser_profile")
        .prefetch_related("advertiser_profile__websites__rap")
        .prefetch_related("advertiser_profile__websites__segment")
    )
    rap_filter_parameter = "rap"

    def filter_by_rap(self, qs):
        rap_key = self.request.query_params.get(self.rap_filter_parameter)
        if not rap_key:
            return qs
        return qs.filter(advertiser_profile__websites__rap=rap_key).distinct()

    def get_queryset(self):
        try:
            from_date, to_date = self.get_date_range_from_request()
        except ValueError:
            return Response(status=HTTP_400_BAD_REQUEST)

        qs = self.daterange_filter(
            from_date, to_date, self.queryset, field="date_joined"
        )
        qs = self.filter_by_rap(qs)

        return qs

CQRS

Command Query Responsibility Separation

Basicly it means: to separate

queries.py

def get_user_by_id(user_id):
    sql = f"select * from user where id={user_id}"
    return execute_sql(sql)

commands.py

def create_user(name, password):
    sql = f"insert into user values(\"{name}\", \"{password}\")"
    execute_sql(sql)

Technicly that's it!

But it is not so simple

Rules of Pure CQRS

  • 1: Command should not return any data
  • 2: Use different models inside and outside CQRS
  • 3: Tests only using Commands and Queries
  • 4: Treat your CQRS as an API

Rule one

Command should not return any data

But what to do when we need to get data after an action?

Use query

def create_user(uuid, name, password):
    sql = f"insert into user values({uuid}, \"{name}\", \"{password}\")"
    execute_sql(sql)
new_uuid = uuid4()

In complicated environment you can never be sure the database instance you are writing is the same you are reading from

Writing to the database can be asynchronous

Rule two

Use different models inside and outside CQRS

Command Query Responsible Separation is about separation of the interior to the exterior.

But what we get in exchange?

Rule three

Tests only using Commands and Queries

def test_create():
    uuid = uuid4()
    save_new(
        {
            "uuid": uuid,
            "name": "user1",
            "email": "user1@my.pl",
            "is_admin": False,
        }
    )
    user = get_by_uuid(uuid)
    assert user.name == "user1"
    assert user.email == "user1@my.pl"
    assert user.is_admin is False

Advantages

  • Change implementation without need to change tests
  • Can check time of our commands and queries
  • Can make the rest of the test not using database

Overall

Advantages

  • Database Agnostic
  • More readable queries
  • Tests separation -> faster unittests

Disadvantages

  • Little more complicated at first
  • For Python developers: no more generic views, no more DRF

When and where to use

  • When you use raw SQL or very complicated ORM queries
  • When you are preparing to switch database
  • When you can not use database in your tests
